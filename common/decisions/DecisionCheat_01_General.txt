	is_shown = {
		has_variable = openCheatDecision_variable
		has_variable = Open_GeneralCheat_Decision_variable
	}

	possible = {

	}

	ai_chance = {
		base = 0
    }
	when_taken = {
		if = {
			limit = {
				any_scope_state = {
					NOT = { has_decree = DecisionCheatDecree_MarkState }
					count = all
				}
			}
			capital = {
				state_region = {
					add_arable_land = 100
				}
			}
		}
		else = {
			every_scope_state = {
				limit = {
					has_decree = DecisionCheatDecree_MarkState
				}
				state_region = {
					add_arable_land = 100
				}
			}
		}
	}
}
